几种常用的排序算法

2019-05-16 · 5 min read
数据结构

选择排序

说明

每一轮遍历选择数组中最小/大的，将其交换到开头。

示例代码

void Sort(int[] _array)
{
    for (int i = 0; i < _array.Length - 1; i++)
    {
        for (int j = i + 1; j < _array.Length; j++)
        {
            if (_array[i] > _array[j])
            {
                int temp = _array[i];
                _array[i] = _array[j];
                _array[j] = temp;
            }
        }
    }
}

冒泡排序

说明

每次比较相邻元素，使小的在前，大的在后

示例代码

void Sort(int[] _array)
{
    for (int i = 0; i < _array.Length - 1; ++i)
    {
        for (int j = 0; j < _array.Length - 1 - i; ++j)
        {
            if (_array[j] > _array[j + 1])
            {
                int temp = _array[j];
                _array[j] = _array[j + 1];
                _array[j + 1] = temp;
            }
        }
    }
}

插入排序

说明

选定元素与前面排好序的元素从后向前进行比较，小于前面元素则与其交换位置，大于等于前面元素则停止当前遍历。

示例代码

void Sort(int[] _array)
{
    for (int i = 1; i < _array.Length; ++i)//只有一个元素视为排序好的
    {
        int current = _array[i];
        int index = i;
        for (int j = i; j > 1 && current < _array[j - 1]; --j)
        {
            index = j - 1;
            _array[j] = _array[j - 1];
        }
        _array[index] = current;
    }
}

希尔排序（缩小增量排序）

说明

在插入排序的基础上，增加了步长的变化。步长由大到1依次遍历进行排序。

示例代码

void Sort(int[] _array)
{
    int length = _array.Length;

    for (int step = length / 2; step > 0; step = step / 2)//步长变化
    {
        for (int startIndex = 0; startIndex < step; ++startIndex)//开始的索引
        {
            for (int i = startIndex + step; i < length; i += step)//根据不同的索引和步长开始插入排序
            {
                int current = _array[i];
                int index = i;
                for (int j = i; j >= step && current < _array[j - step]; j -= step)
                {
                    index = j - step;
                    _array[j] = _array[j - step];
                }
                _array[index] = current;
            }
        }
    }
}

归并排序

说明

需要额外的存储空间来进行缓存。将数组进行二分，直至最小。每组中的元素总是小的先放入缓存数组，大的后放入缓存数组，以完成排序。

示例代码

void Sort5(int[] _array, int _startIndex, int _endIndex, int[] _tempArray)//为了避免不停new数组，传入了一个缓存数组进来
{
    if (_startIndex >= _endIndex) return;
    int middle = (_startIndex + _endIndex) / 2;
    Sort5(_array, _startIndex, middle, _tempArray);
    Sort5(_array, middle + 1, _endIndex, _tempArray);

    int i = _startIndex;
    int j = middle + 1;
    int k = i;
    while (i <= middle && j <= _endIndex)
    {
        if (_array[i] < _array[j])//两组比较，小的先放
        {
            _tempArray[k] = _array[i];
            i++;
        }
        else
        {
            _tempArray[k] = _array[j];
            j++;
        }
        k++;
    }
    while (i <= middle)//剩下的左面组里元素肯定小于右面组里元素，所以先放左面组
    {
        _tempArray[k] = _array[i];
        i++;
        k++;
    }
    while (j <= _endIndex)
    {
        _tempArray[k] = _array[j];
        j++;
        k++;
    }

    for (k = _startIndex; k <= _endIndex; k++)//将缓存数组里的数据填充到实际数组中
    {
        _array[k] = _tempArray[k];
    }
}

快速排序

说明

选择一个元素作为基本元素，使元素左面的元素均小于它，元素右面的元素均大于它。之后将元素左面和元素右面的数组分别进行此操作，直至排序完成。
##示例代码

void Sort6(int[] _array, int _startIndex, int _endIndex)//选定_startIndex位置元素为基准元素
{
    if (_startIndex >= _endIndex) return;
    if (_endIndex >= _array.Length) return;
    int i = _startIndex;
    int j = _endIndex;
    while (i < j)
    {
        while (_array[j] >= _array[i] && i < j)
        {
            j--;
        }
        if (_array[j] < _array[i])//右面元素小于基准元素，交换基准元素和右面元素位置。
        {
            int temp = _array[j];
            _array[j] = _array[i];
            _array[i] = temp;
        }

        while (_array[i] <= _array[j] && i < j)
        {
            i++;
        }
        if (_array[i] > _array[j])//左面元素大于基准元素，交换基准元素位置
        {
            int temp = _array[j];
            _array[j] = _array[i];
            _array[i] = temp;
        }
    }
    Sort6(_array, _startIndex, i - 1);
    Sort6(_array, j + 1, _endIndex);
}
